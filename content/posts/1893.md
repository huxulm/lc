# 差分写法
 
> [原题 1893.检查是否区域内所有整数都被覆盖](https://leetcode.cn/problems/check-if-all-the-integers-in-a-range-are-covered)
```go
func isCovered(a [][]int, l, r int) bool {
	d := [52]int{}
	for _, p := range a {
		d[p[0]]++
		d[p[1]+1]--
	}
	s := 0
	for _, v := range d[:l] {
		s += v
	}
	for _, v := range d[l : r+1] {
		if s += v; s == 0 {
			return false
		}
	}
	return true
}
```
