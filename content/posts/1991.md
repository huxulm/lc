# Go 模拟
 
> [原题 1991.找到数组的中间位置](https://leetcode.cn/problems/find-the-middle-index-in-array)
```go
func findMiddleIndex(a []int) int {
	tot := 0
	for _, v := range a {
		tot += v
	}
	s := 0
	for i, v := range a {
		if s*2 == tot-v {
			return i
		}
		s += v
	}
	return -1
}
```
