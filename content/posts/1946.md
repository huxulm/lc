# Go 贪心
 
> [原题 1946.子字符串突变后可能得到的最大整数](https://leetcode.cn/problems/largest-number-after-mutating-substring)
```go
func maximumNumber(num string, change []int) string {
	s := []byte(num)
	i, n := 0, len(s)
	for ; i < n && byte(change[s[i]&15]) <= s[i]&15; i++ {
	}
	for ; i < n && byte(change[s[i]&15]) >= s[i]&15; i++ {
		s[i] = '0' + byte(change[s[i]&15])
	}
	return string(s)
}
```
