# Go 模拟
 
> [原题 1941.检查是否所有字符出现次数相同](https://leetcode.cn/problems/check-if-all-characters-have-equal-number-of-occurrences)
```go
func areOccurrencesEqual(s string) bool {
	cnt := [26]int{}
	for _, b := range s {
		cnt[b-'a']++
	}
	target := 0
	for _, c := range cnt {
		if c == 0 {
			continue
		}
		if target == 0 {
			target = c
		} else if c != target {
			return false
		}
	}
	return true
}
```
