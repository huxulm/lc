# 遍历，找第一个回文串
 
> [原题 2108.找出数组中的第一个回文字符串](https://leetcode.cn/problems/find-first-palindromic-string-in-the-array)
```go
func firstPalindrome(words []string) string {
next:
	for _, w := range words {
		for i, n := 0, len(w); i < n/2; i++ {
			if w[i] != w[n-1-i] {
				continue next
			}
		}
		return w
	}
	return ""
}
```
