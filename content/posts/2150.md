# 哈希表模拟
 
> [原题 2150.找出数组中的所有孤独数字](https://leetcode.cn/problems/find-all-lonely-numbers-in-the-array)
```go
func findLonely(nums []int) (ans []int) {
	cnt := map[int]int{}
	for _, v := range nums {
		cnt[v]++
	}
	for v, c := range cnt {
		if c == 1 && cnt[v+1] == 0 && cnt[v-1] == 0 {
			ans = append(ans, v)
		}
	}
	return
}
```
