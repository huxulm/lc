# 贪心+上取整
 
> [原题 1936.新增的最少台阶数](https://leetcode.cn/problems/add-minimum-number-of-rungs)
```go
func addRungs(rungs []int, dist int) (ans int) {
	pre := 0
	for _, h := range rungs {
		if d := h - pre; d > dist {
			ans += (d - 1) / dist // 等价于 ceil(d / dist) - 1
		}
		pre = h
	}
	return
}
```
