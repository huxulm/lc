# Go 统计每个元素的出现次数
 
> [原题 1940.排序数组之间的最长公共子序列](https://leetcode.cn/problems/longest-common-subsequence-between-sorted-arrays)
```go
func longestCommonSubsequence(arrays [][]int) (ans []int) {
	cnt := [101]int{}
	for _, a := range arrays {
		for _, v := range a {
			cnt[v]++
		}
	}
	for v, c := range cnt {
		if c == len(arrays) {
			ans = append(ans, v)
		}
	}
	return
}
```
